src.static_air_site_generation

A module to process raw CSV Mott MacDonald supplied site data. Author: Ross Cochrane

 1"""
 2A module to process raw CSV Mott MacDonald supplied site data.
 3Author: Ross Cochrane
 4"""
 5
 6import pandas as pd
 7import json
 8from pyproj import Transformer
 9
10
11
12# Initialize transformer from EPSG:27700 (OSGB) to WGS84
13transformer = Transformer.from_crs("epsg:27700", "epsg:4326", always_xy=True)
14
15# Read CSV
16df = pd.read_csv("data/AIRQUALITY_DEFINITION.csv", usecols=[
17    "SYSTEMCODENUMBER", "SHORTDESCRIPTION", "LONGDESCRIPTION",
18    "NORTHING", "EASTING", "LASTUPDATED"
19])
20
21# Drop rows with incomplete coordinates
22df = df.dropna(subset=["NORTHING", "EASTING"])
23
24# Sort and re-index system code numbers
25df = df.sort_values(by="SYSTEMCODENUMBER").reset_index(drop=True)
26
27# Generate SITE IDs
28df["systemCodeNumber"] = ["SITE{:03d}".format(i + 1) for i in range(len(df))]
29
30# Convert northing/easting to lat/lon
31def convert_coords(e, n):
32    lon, lat = transformer.transform(e, n)
33    return lat, lon
34
35fixed_timestamp = "2025-06-26T19:47:00.000+0000"
36
37json_output = []
38for _, row in df.iterrows():
39    lat, lon = convert_coords(row["EASTING"], row["NORTHING"])
40    entry = {
41        "systemCodeNumber": row["systemCodeNumber"],
42        "definitions": [
43            {
44                "longDescription": row["LONGDESCRIPTION"],
45                "point": {
46                    "easting": row["EASTING"],
47                    "northing": row["NORTHING"],
48                    "latitude": round(lat, 6),
49                    "longitude": round(lon, 6)
50                },
51                "lastUpdated": fixed_timestamp
52            }
53        ]
54    }
55    json_output.append(entry)
56
57
58# Save to JSON file
59with open("data/metadata.json", "w") as f:
60    json.dump(json_output, f, indent=2)
